## Installation

To install the swiper component, run the following command:

```bash
npx shadcn@latest add @fulldev/swiper
```

## Usage

```ts
import {
  Swiper,
  SwiperContent,
  SwiperNext,
  SwiperPagination,
  SwiperPrevious,
  SwiperSlide,
} from "@/components/ui/swiper"
```

```astro
<Swiper
  options={{
    loop: true,
    autoplay: {
      enabled: true,
      delay: 3500,
      disableOnInteraction: false,
    },
  }}
>
  <SwiperContent>
    <SwiperSlide>
      <div>Slide 1</div>
    </SwiperSlide>
    <SwiperSlide>
      <div>Slide 2</div>
    </SwiperSlide>
  </SwiperContent>
  <SwiperPrevious />
  <SwiperNext />
  <SwiperPagination />
</Swiper>
```
